\vspace{0.4cm}
\textbf{Rôle de chaque couleur :}
\begin{itemize}
  \item \textbf{Bleu nuit (2C3E50)} : en-tête, pied de page, en-têtes de tableau, texte principal.
  \item \textbf{Turquoise (1ABC9C)} : couleur d'accent — liens actifs, boutons, valeurs des statistiques.
  \item \textbf{Turquoise foncé (16A085)} : survol des boutons.
  \item \textbf{Bleu (2980B9)} : titres de section.
  \item \textbf{Bleu très clair (F0F7FC)} : survol des lignes du tableau.
  \item \textbf{Gris clair (F4F6F8)} : fond général des pages.
  \item \textbf{Blanc (FFFFFF)} : fond des cartes et des tableaux.
\end{itemize}
